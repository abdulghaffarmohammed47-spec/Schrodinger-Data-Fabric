```python
from cryptography.hazmat.primitives.asymmetric import rsa, padding
from cryptography.hazmat.primitives import serialization, hashes
from cryptography.fernet import Fernet

# Generate RSA keys
def generate_keys():
    private_key = rsa.generate_private_key(public_exponent=65537, key_size=2048)

    public_key = private_key.public_key()

    return private_key, public_key


def serialize_private(private_key):
    return private_key.private_bytes(
        encoding=serialization.Encoding.PEM,
        format=serialization.PrivateFormat.PKCS8,
        encryption_algorithm=serialization.NoEncryption()
    )


def serialize_public(public_key):
    return public_key.public_bytes(
        encoding=serialization.Encoding.PEM,
        format=serialization.PublicFormat.SubjectPublicKeyInfo
    )


def load_private(pem):
    return serialization.load_pem_private_key(pem, password=None)


def load_public(pem):
    return serialization.load_pem_public_key(pem)


# AES (via Fernet)
def generate_symmetric_key():
    return Fernet.generate_key()


def encrypt_data(data, key):
    return Fernet(key).encrypt(data)


def decrypt_data(data, key):
    return Fernet(key).decrypt(data)


# Encrypt symmetric key with RSA
def encrypt_key(sym_key, public_key):
    return public_key.encrypt(
        sym_key,
        padding.OAEP(mgf=padding.MGF1(algorithm=hashes.SHA256()), algorithm=hashes.SHA256(), label=None)
    )


def decrypt_key(enc_key, private_key):
    return private_key.decrypt(
        enc_key,
        padding.OAEP(mgf=padding.MGF1(algorithm=hashes.SHA256()), algorithm=hashes.SHA256(), label=None)
    )
```

# 📦 STEP 1: CREATE BEHAVIOR MODEL

## backend/behavior_ml.py

```python
import time
import numpy as np
from collections import defaultdict, deque

# Store user behavior
USER_SEQUENCES = defaultdict(lambda: {
    "timestamps": deque(maxlen=10),
    "vectors": deque(maxlen=10)
})


def update_behavior(user_id, vector):
    now = time.time()

    USER_SEQUENCES[user_id]["timestamps"].append(now)
    USER_SEQUENCES[user_id]["vectors"].append(vector)


def compute_features(user_id):
    data = USER_SEQUENCES[user_id]

    if len(data["timestamps"]) < 3:
        return None  # not enough data

    # time gaps
    gaps = np.diff(list(data["timestamps"]))

    avg_gap = np.mean(gaps)
    std_gap = np.std(gaps)

    # vector behavior
    vectors = np.array(data["vectors"])
    avg_vec = np.mean(vectors, axis=0)

    return {
        "avg_gap": avg_gap,
        "std_gap": std_gap,
        "avg_vec": avg_vec
    }


def anomaly_score(user_id, current_vector):
    features = compute_features(user_id)

    if not features:
        return 0  # treat as normal initially

    last_time = USER_SEQUENCES[user_id]["timestamps"][-1]
    gap = time.time() - last_time

    # z-score for timing
    if features["std_gap"] == 0:
        time_score = 0
    else:
        time_score = abs((gap - features["avg_gap"]) / features["std_gap"])

    # vector deviation
    vec_score = np.linalg.norm(current_vector - features["avg_vec"])

    return time_score + vec_score
```

```python
from fastapi import FastAPI, UploadFile
import os
import uuid

app = FastAPI()

STORAGE = "node_storage"
os.makedirs(STORAGE, exist_ok=True)

@app.post("/store")
async def store(file: UploadFile):
    data = await file.read()
    shard_id = str(uuid.uuid4())

    path = os.path.join(STORAGE, shard_id)
    with open(path, "wb") as f:
        f.write(data)

    return {"shard_id": shard_id}


@app.get("/get/{shard_id}")
def get_shard(shard_id: str):
    path = os.path.join(STORAGE, shard_id)

    if not os.path.exists(path):
        return {"error": "not found"}

    with open(path, "rb") as f:
        return f.read()
```

```python
import requests
from node_registry import pick_nodes

def store_shard(data):
    nodes = pick_nodes()
    locations = []

    for node in nodes:
        try:
            res = requests.post(f"{node}/store", files={"file": data}, timeout=2)
            shard_id = res.json()["shard_id"]
            locations.append((node, shard_id))
        except:
            continue

    return locations


def fetch_shard(locations):
    for node, shard_id in locations:
        try:
            res = requests.get(f"{node}/get/{shard_id}", timeout=2)
            if res.status_code == 200:
                return res.content
        except:
            continue

    return None
```

```python
from logger import get_logs

def compute_metrics():
    logs = get_logs()

    total = len(logs)
    attacks = sum(1 for l in logs if l["type"] == "DECOY_TRIGGERED")
    success = sum(1 for l in logs if l["type"] == "ACCESS_GRANTED")

    return {
        "total_events": total,
        "attacks_detected": attacks,
        "successful_access": success,
        "attack_rate": attacks / total if total else 0
    }
```
